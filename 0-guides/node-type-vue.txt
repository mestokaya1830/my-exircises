Vite Proxy Setup (vite.config.ts):

⚙️ Configured to point directly to your Express server.

TypeScript
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'

export default defineConfig({
  plugins: [vue()],
  server: {
    proxy: {
      '/api': {
        target: 'http://localhost:3000',
        changeOrigin: true,
      }
    }
  },
})
Vue Component Logic (App.vue):

Code snippet
<template>
  <h1>{{ title }}</h1>
  <pre>{{ user }}</pre>
</template>

<script>
import axios from "axios";
export default {
  name: "typescript",
  data() {
    return {
      title: "Type-Vue-Node",
      user: {},
    };
  },
  mounted() {
    this.getUser();
  },
  methods: {
    async getUser() {
      await axios.get("/api").then((result) => {
        this.user = result.data;
      });
    },
  },
};
</script>

Backend Logic (src/index.ts):

TypeScript
import express, { Request, Response, NextFunction } from 'express';
import { ErrorRequestHandler } from "express";
import cors from 'cors';
import dotenv from 'dotenv'
dotenv.config()

const app = express();

app.use(cors());
app.use(express.json());

app.get('/', (req: Request, res: Response) => {
  console.log('test')
  const user: {id: number, full_name: string, age: number, active: boolean }[] = [
    { id: 0, full_name: 'Mustafa Kaya', age: 45, active: true },
    { id: 1, full_name: 'Ekrem Kaya', age: 23, active: true }
  ]
  res.json(user);
});

// 404 Handler
app.use((req: Request, res: Response) => {
  res.status(404).json({ error: "404 Page Not Found" });
});

// Error Handler
const errorHandler: ErrorRequestHandler = (err, req, res, next) => {
  console.error(err);
  res.status(500).json({
    success: false,
    message: err.message || "Internal Server Error"
  });
};
app.use(errorHandler);

// Server Start
const start = async () => {
  const PORT = Number(process.env.PORT) || 3000
  app.listen(PORT, () => {
    console.log('Server working at port', PORT)
  })
}

start()
