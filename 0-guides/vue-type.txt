{
  "compilerOptions": {

    "module": "esnext",            // Vite ESM kullanır
    "target": "esnext",            // Modern tarayıcılar için
    "moduleResolution": "bundler", // Vite uyumlu modül çözümlemesi
    "lib": ["esnext", "dom", "dom.iterable"], // Tarayıcı API'leri + modern JS
    "types": ["vite/client"],      // import.meta.env, ?url, ?raw gibi Vite tiplerini tanı

    "jsx": "preserve",        // Vue'nun kendi JSX dönüşümüne bırak

    "strict": true,                     // Tüm katı kontrolleri etkinleştir
    "noUncheckedIndexedAccess": true,   // arr[0] → T | undefined olarak değerlendir

    "verbatimModuleSyntax": true, // import type kullanımını zorunlu kıl
    "isolatedModules": true,      // Her dosyayı bağımsız derle (Vite uyumu)
    "moduleDetection": "force",   // Her dosyayı otomatik modül say
    "skipLibCheck": true          // node_modules .d.ts hatalarını atla

  },
  "include": [
    "src/**/*.ts",
    "src/**/*.vue",  // .vue dosyalarını dahil et

    "node_modules",
    "dist"
  ]
